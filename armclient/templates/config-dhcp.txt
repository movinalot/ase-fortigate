#cloud-config
runcmd:
config system global
set hostname fgt-ase-3nics-dhcp
set admintimeout 120
set gui-theme mariner
end
config router static
 edit 1
 set device port1
 set dynamic-gateway enable
 next
end
config system interface
edit port1
set alias mgmt
set mode dhcp
set allowaccess ping https ssh fgfm
next
edit port2
set alias wan
set mode static
set ip 192.168.6.10/24
set allowaccess ping https ssh fgfm
next
edit port3
set alias lan
set mode static
set ip 192.168.5.10/24
set allowaccess ping
next
end
config system admin
edit fgtaseadmin
set accprofile super_admin
set password Fortinet1234!
set vdom root
next
end
